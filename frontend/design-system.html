07 · CSS Variables

토큰 명세 · :root

단계 1 적용 시 frontend/index.html :root 에 그대로 추가. 기존 토큰 (--bg/--acc/--t1 등) 은 호환을 위해 alias 로 매핑.

:root {
  /* Background scale */
  --bg-base:        #0A0A0F;
  --bg-elev-1:      #0F0F15;
  --bg-elev-2:      #14141C;
  --bg-elev-3:      #1A1A24;

  /* Glass surfaces */
  --glass-1:        rgba(255,255,255,0.035);
  --glass-2:        rgba(255,255,255,0.06);
  --glass-3:        rgba(255,255,255,0.085);

  /* Accent · Electric Orange */
  --accent-500:     #FF6B35;  /* primary */
  --accent-600:     #F25218;  /* hover */
  --accent-700:     #C84B18;  /* legacy brand */

  /* Semantic */
  --positive:       #4ADE80;
  --warning:        #FCD34D;
  --negative:       #F87171;

  /* Text scale */
  --text-primary:   #F4F4F5;
  --text-secondary: #D4D4D8;
  --text-tertiary:  #A1A1AA;
  --text-quiet:     #71717A;

  /* Typography */
  --font-display:   'Sora', 'Pretendard Variable', sans-serif;
  --font-body:      'Pretendard Variable', system-ui, sans-serif;
  --font-mono:      'JetBrains Mono', monospace;

  /* Spacing · 4pt grid */
  --space-1: 4px;  --space-2: 8px;  --space-3: 12px;
  --space-4: 16px; --space-6: 24px; --space-8: 32px;
  --space-12: 48px; --space-16: 64px;

  /* Radius */
  --radius-sm: 6px; --radius-md: 10px; --radius-lg: 14px;
  --radius-xl: 18px; --radius-2xl: 24px;

  /* Shadow + glow */
  --shadow-md:   0 1px 0 rgba(255,255,255,0.05) inset, 0 8px 24px rgba(0,0,0,0.4);
  --glow-orange: 0 0 24px rgba(255,107,53,0.35);

  /* Motion */
  --duration-fast:   150ms;
  --duration-normal: 250ms;
  --ease-out:        cubic-bezier(0.16, 1, 0.3, 1);

  /* Backdrop blur */
  --blur-md:  blur(20px) saturate(1.6);
  --blur-lg:  blur(40px) saturate(1.8);

  /* Legacy alias (호환용 — 점진적 마이그레이션) */
  --bg:    var(--bg-base);
  --s0:    var(--bg-elev-1);
  --s1:    var(--bg-elev-2);
  --t1:    var(--text-primary);
  --t2:    var(--text-secondary);
  --t3:    var(--text-tertiary);
  --acc:   var(--accent-500);
  --accD:  var(--accent-600);
}
