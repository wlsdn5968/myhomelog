08 · CSS Variables

토큰 명세 · :root

:root {
  /* Background scale (light) */
  --bg-base:        #FFFFFF;
  --bg-elev-1:      #FAFAFA;
  --bg-elev-2:      #F5F5F5;
  --bg-elev-3:      #EFEFEF;

  /* Glass surfaces (black-based on white) */
  --glass-1:        rgba(0,0,0,0.025);
  --glass-2:        rgba(0,0,0,0.05);
  --glass-3:        rgba(0,0,0,0.08);

  /* Accent · 변경 없음 */
  --accent-500:     #FF6B35;
  --accent-700:     #C84B18;  /* legacy brand */
  --accent-shadow:  0 4px 12px rgba(255,107,53,0.18), 0 0 0 1px rgba(255,107,53,0.10);

  /* Text scale (dark on light) */
  --text-primary:   #0A0A0F;
  --text-secondary: #404048;
  --text-tertiary:  #5C5749;  /* F-1 일치 · WCAG AA 5.04:1 */

  /* Typography */
  --font-display:   'Sora', 'Pretendard Variable', sans-serif;
  --font-body:      'Pretendard Variable', system-ui, sans-serif;
  --font-mono:      'JetBrains Mono', monospace;

  /* Shadow · 라이트 ambient depth */
  --shadow-md: 0 1px 3px rgba(15,23,42,0.05), 0 8px 24px rgba(15,23,42,0.06);

  /* Spacing · Radius · Motion · Backdrop blur 모두 다크와 동일 */

  /* Legacy alias — 점진 마이그레이션 */
  --bg: var(--bg-base);
  --s0: var(--bg-elev-1);
  --t1: var(--text-primary);
  --t3: var(--text-tertiary);  /* F-1 작업 자동 적용 */
  --acc: var(--accent-500);
}
